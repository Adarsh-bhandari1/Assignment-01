void twosum(int a[] , int n)
{
    int i,j,t;
    printf("Enter the target value \n ");
    scanf("%d",&t);
    for(i=0;i<n-1;i++)
    {
        for(j=i+1;j<=n;j++)
        {
            if(a[i]+a[j]==t)
            {
                printf("The ans is %d %d",i,j);
            }

        }
    }
}
